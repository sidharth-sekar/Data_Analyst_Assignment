-- CLINIC QUERY SOLUTIONS (PART B)
-- ==========================================

-- Q1: Revenue from each sales channel
SELECT sales_channel,
       SUM(amount) AS total_revenue
FROM clinic_sales
WHERE YEAR(datetime) = 2021
GROUP BY sales_channel;


-- Q2: Top 10 valuable customers
SELECT uid,
       SUM(amount) AS total_spent
FROM clinic_sales
WHERE YEAR(datetime) = 2021
GROUP BY uid
ORDER BY total_spent DESC
LIMIT 10;


-- Q3: Month-wise revenue and expense
SELECT MONTH(cs.datetime) AS month_no,
       SUM(cs.amount) AS revenue,
       SUM(e.amount) AS expense,
       SUM(cs.amount) - SUM(e.amount) AS profit
FROM clinic_sales cs
JOIN expenses e
ON MONTH(cs.datetime) = MONTH(e.datetime)
WHERE YEAR(cs.datetime) = 2021
GROUP BY MONTH(cs.datetime);


-- Q4: Most profitable clinic per city
SELECT c.city,
       c.clinic_name,
       SUM(cs.amount) - COALESCE(SUM(e.amount),0) AS profit
FROM clinics c
JOIN clinic_sales cs
ON c.cid = cs.cid
LEFT JOIN expenses e
ON c.cid = e.cid
WHERE MONTH(cs.datetime) = 9
GROUP BY c.city, c.clinic_name
ORDER BY c.city, profit DESC;


-- Q5: Second least profitable clinic per state
SELECT c.state,
       c.clinic_name,
       SUM(cs.amount) - COALESCE(SUM(e.amount),0) AS profit
FROM clinics c
JOIN clinic_sales cs
ON c.cid = cs.cid
LEFT JOIN expenses e
ON c.cid = e.cid
WHERE MONTH(cs.datetime) = 9
GROUP BY c.state, c.clinic_name
ORDER BY c.state, profit ASC;
